const BOOK_PRICE = 800;
const DISCOUNT_PER_COUNT = { 1: 0, 2: 0.05, 3: 0.10, 4: 0.20, 5: 0.25 };

/**
    * Sort all books by series.
    * @param {array} books - The original array of books to sort
    * @param {array} sortedBooks - The book array once sorted
    * @param {number} serieIndex -  The index to add current sorted book's series 
    */
function splitBooks(books, sortedBooks, serieIndex = 0) {
    let set = new Set(books);

    set.forEach((sortedBook) =>  {
        books.splice(books.indexOf(sortedBook), 1);
    });

    sortedBooks[serieIndex] = [...set];
    
    if (books.length === 0) return sortedBooks;

    else return splitBooks(books, sortedBooks, ++serieIndex);
}

function sortForBestDiscount(sortedBooks) {
    sortedBooks.forEach((batch) => {
        if (batch.length === 5) {
            sortedBooks.forEach((compareBatch) => {
                if (compareBatch.length === 3) {
                    compareBatch.push(batch.pop());
                }
            });
        }
    });
    return sortedBooks;
}

function computePrice(sortedBooks, bookPrice) {
    return sortedBooks.reduce((accumulator, batch) => {
        let discount = DISCOUNT_PER_COUNT[batch.length];
        return accumulator + (batch.length * bookPrice) * (1 - discount);
    }, 0);
}

export const cost = (books) => {
    if (books.length === 0) return 0;
    return computePrice(sortForBestDiscount(splitBooks(books, [])), BOOK_PRICE);
}
